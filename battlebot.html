About

The Build

dEaTh wEdGe is a combat wedge robot built by Zarf and Hitansh. It's controlled over ESP-NOW from a two-axis joystick remote, driven by a dual motor setup on a custom chassis.

The remote reads the joystick, filters out drift and noise, and fires off a command packet 50 times a second. The car mixes those into left/right motor speeds and drives — and if the signal ever drops for half a second, it stops dead rather than run blind.

How It Works

#include <esp_now.h>
#include <WiFi.h>

// Use these pins — more stable ADC on ESP32
#define JOY_LR  32
#define JOY_FB  33

uint8_t receiverMAC[] = {0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF};

typedef struct {
  int lr;
  int fb;
} JoyData;

JoyData data;
esp_now_peer_info_t peerInfo;

int applyDeadzone(int val, int dz = 30) {  // increased deadzone
  return (abs(val) < dz) ? 0 : val;
}

void onSent(const uint8_t *mac, esp_now_send_status_t status) {
  // Serial.println(status == ESP_NOW_SEND_SUCCESS ? "OK" : "FAIL");
}

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);
  WiFi.disconnect();

  if (esp_now_init() != ESP_OK) {
    Serial.println("ESP-NOW init failed!");
    while (true);
  }

  esp_now_register_send_cb(onSent);

  memcpy(peerInfo.peer_addr, receiverMAC, 6);
  peerInfo.channel = 0;
  peerInfo.encrypt = false;

  if (esp_now_add_peer(&peerInfo) != ESP_OK) {
    Serial.println("Failed to add peer!");
    while (true);
  }

  Serial.println("Transmitter ready!");
  Serial.print("MAC: "); Serial.println(WiFi.macAddress());
}

void loop() {
  // Average 5 readings to reduce ADC noise
  long rawLR = 0, rawFB = 0;
  for (int i = 0; i < 5; i++) {
    rawLR += analogRead(JOY_LR);
    rawFB += analogRead(JOY_FB);
  }
  rawLR /= 5;
  rawFB /= 5;

  data.lr = applyDeadzone(map(rawLR, 0, 4095, -255, 255));
  data.fb = applyDeadzone(map(rawFB, 0, 4095, -255, 255));

  // Debug — uncomment to check values
  // Serial.printf("LR: %d  FB: %d\n", data.lr, data.fb);

  esp_now_send(receiverMAC, (uint8_t *)&data, sizeof(data));
  delay(20);
}

#include <esp_now.h>
#include <WiFi.h>

#define IN1 26
#define IN2 27
#define ENA 25
#define IN3 14
#define IN4 12
#define ENB 33

#define CH_LEFT  0
#define CH_RIGHT 1
#define PWM_FREQ 20000  // 20kHz — silent!
#define PWM_RES  8
#define TIMEOUT_MS 500

unsigned long lastPacket = 0;

typedef struct {
  int lr;
  int fb;
} JoyData;

JoyData incoming;

void setLeft(int spd) {
  if (spd > 0)      { digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW); }
  else if (spd < 0) { digitalWrite(IN1, LOW);  digitalWrite(IN2, HIGH); }
  else              { digitalWrite(IN1, LOW);   digitalWrite(IN2, LOW); }
  ledcWrite(CH_LEFT, abs(spd));
}

void setRight(int spd) {
  if (spd > 0)      { digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW); }
  else if (spd < 0) { digitalWrite(IN3, LOW);  digitalWrite(IN4, HIGH); }
  else              { digitalWrite(IN3, LOW);   digitalWrite(IN4, LOW); }
  ledcWrite(CH_RIGHT, abs(spd));
}

void stopMotors() {
  digitalWrite(IN1,LOW); digitalWrite(IN2,LOW);
  digitalWrite(IN3,LOW); digitalWrite(IN4,LOW);
  ledcWrite(CH_LEFT,  0);
  ledcWrite(CH_RIGHT, 0);
}

void drive(int fb, int lr) {
  int leftSpd  = constrain(fb + lr, -255, 255);
  int rightSpd = constrain(fb - lr, -255, 255);

  // Boost: scale up so full joystick = full speed always
  int maxVal = max(abs(leftSpd), abs(rightSpd));
  if (maxVal > 0 && (abs(fb) == 255 || abs(lr) == 255)) {
    leftSpd  = leftSpd  * 255 / maxVal;
    rightSpd = rightSpd * 255 / maxVal;
  }

  setLeft(leftSpd);
  setRight(rightSpd);
}

void onReceive(const uint8_t *mac, const uint8_t *incomingData, int len) {
  if (len != sizeof(JoyData)) return;
  memcpy(&incoming, incomingData, sizeof(JoyData));
  lastPacket = millis();
  drive(incoming.fb, incoming.lr);
}

void setup() {
  Serial.begin(115200);
  pinMode(IN1,OUTPUT); pinMode(IN2,OUTPUT);
  pinMode(IN3,OUTPUT); pinMode(IN4,OUTPUT);

  ledcSetup(CH_LEFT,  PWM_FREQ, PWM_RES);
  ledcSetup(CH_RIGHT, PWM_FREQ, PWM_RES);
  ledcAttachPin(ENA, CH_LEFT);
  ledcAttachPin(ENB, CH_RIGHT);
  stopMotors();

  WiFi.mode(WIFI_STA);
  WiFi.disconnect();

  if (esp_now_init() != ESP_OK) {
    Serial.println("ESP-NOW init failed!");
    while (true);
  }

  esp_now_register_recv_cb(onReceive);
  Serial.println("Receiver ready!");
  Serial.print("MAC: "); Serial.println(WiFi.macAddress());
}

void loop() {
  if (millis() - lastPacket > TIMEOUT_MS) {
    stopMotors();
  }
}
